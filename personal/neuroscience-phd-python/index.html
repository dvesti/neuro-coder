import matplotlib.pyplot as plt
from matplotlib.patches import Wedge
def make_phd_plot(data):
    # Ignoring data
    ignore_data(data)

    #plot results
    main = plt.Circle((0,0),5, fill=False)
    angle1, angle2 = 0, 180
